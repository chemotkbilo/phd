\def\subsectionnumberingsuffix@cx{}% 
\def\subsectionnumberingprefix@cx{}%
\cxset{%
  subsection numbering suffix/.store in=\subsectionnumberingsuffix@cx, 
  %alias
  section number after/.store in =\sectionnumberafter@cx,
  subsection numbering prefix/.store in=\subsectionnumberingprefix@cx,
}

\cxset{  
  subsection numbering/.is choice,
  subsection numbering/roman/.code={%
       \gdef\thesection{\subsectionnumberingprefix@cx\@roman\c@section}%
          \renewsection},
 subsection numbering/Roman/.code={%
       \gdef\thesection{\subsectionnumberingprefix@cx\@Roman\c@section}%
          \renewsection},
  subsection numbering/(roman)/.code={%
       \gdef\thesection{\subsectionnumberingprefix@cx(\@roman\c@section)}%
       \renewsection},
  subsection numbering/(Roman)/.code={%
       \gdef\thesection{\subsectionnumberingprefix@cx(\@Roman\c@section)}%
          \renewsection},
  subsection numbering/arabic/.code={%
       \gdef\thesection{\subsectionnumberingprefix@cx\@arabic\c@section\subsectionnumberingsuffix@cx}%
          \renewsection},
  subsection numbering/numeric/.code={%
       \gdef\thesection{\subsectionnumberingprefix@cx\@arabic\c@section\subsectionnumberingsuffix@cx}%
          \renewsection},
  subsection numbering/none/.code={\gdef\thesection{\hspace*{-1em}}\renewsection},
  subsection numbering/alpha/.code={\gdef\thesection{\alphalph\c@section}},
  subsection numbering/Alpha/.code={\gdef\thesection{\AlphAlph\c@section}},
  subsection numbering/words/.code={\gdef\thesection{\subsectionnumberingprefix@cx%
                                                   \words@cx{\@arabic\c@section}}},
  subsection numbering/Words/.code={%
 \gdef\thesection{\subsectionnumberingprefix@cx\words@cx{\@arabic\c@section}}},
   subsection numbering/WORDS/.code={\gdef\thesection{\subsectionnumberingprefix@cx \words@cx{\@arabic\c@section}}},
}   
%    \end{macrocode}
