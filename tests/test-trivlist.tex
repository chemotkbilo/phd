% Example of usage of trivlist environment
% http://tex.stackexchange.com/questions/47057/use-of-trivlist-and-list-in-defining-environments
\documentclass{article}
\usepackage{lipsum}
\begin{document}
\makeatletter
\newenvironment{name}
  { \trivlist\itemsep100pt\item
   \tabular{@{}ll@{}}}
  {\endtabular\endtrivlist}
  
\lipsum[1]  
\begin{name}
   First    & Mary  \\
   Second   & Jones \\
   Nickname & --- \\
\end{name}

\begin{name}
   First    & Mary  \\
   Second   & Jones \\
   Nickname & --- \\
\end{name}
\lipsum[2]

\tabular{@{}ll@{}}
   First    & Mary  \\
   Second   & Jones \\
   Nickname & --- \\
\endtabular



\end{document}
