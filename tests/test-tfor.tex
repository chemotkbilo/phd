\newif\if@hascomma
\@hascommatrue
\def\hascomma#1{%
    \@hascommafalse
    \@tfor\next:=#1\do{%
    \edef\tempa{\next}%
    \edef\tempb{,}%
    {\if\tempa\tempb
         \global\@hascommatrue
    \fi}%
}}%


\def\checkforcomma@cx#1{%
\hascomma{#1}%
\if@hascomma%
    \expandafter\@firstoftwo%
  \else%
    \expandafter\@secondoftwo% 
\fi
}
